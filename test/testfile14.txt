int getint() {
	int x = 0, c = 0, minus = 0;
	while((c < 48 || c > 57) && c != 45) {
		c = getchar();
	}
	if (c == 45) {
		minus = 1;
		c = getchar();
	}
	while(c >= 48 && c <= 57) {
		x = x * 10 + c - 48;
		c = getchar();
	}
	if (minus) {
		x = -x;
	}
	return x;
}
int __fact(int x, int r) {
    if (x <= 1) return r;
    else return __fact(x - 1, (r * x) % 19260817);
    return 0;
}

int fact(int x) {
    return __fact(x, 1);
}
